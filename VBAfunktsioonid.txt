  korruta2 = arv * 2
End Function

Function kolmnurkPindala(alus, korgus As integer) As double
    Const k = 1 / 2
    kolmnurkPindala = alus * korgus * k
End Function

Function esitahtVaiksena(sona As string) As string
  esitahtVaiksena = LCase(Mid(sona, 1, 1))
End Function
